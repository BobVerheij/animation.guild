Example One: Mouse Follow

code stack: html, css, javascript, P5js

Code

var exampleOne = function (p) { var x; var y; var target; var current; var sizeTarget; var sizeCurrent; var speed = 2.5; p.setup = function () { var thisDiv = document.getElementById("exampleTwo"); var xSize = thisDiv.offsetWidth; var ySize = thisDiv.offsetHeight; p.createCanvas(xSize, ySize); x = p.width / 2; y = p.height / 2; target = p.createVector(x, y); current = p.createVector(x, y); sizeCurrent = 0; sizeTarget = 0; }; p.draw = function () { p.background(255); p.fill(1); if (!(p.mouseX < 0 || p.mouseX> p.width || p.mouseY < 0 || p.mouseY> p.height)) { target.x = p.mouseX; target.y = p.mouseY; sizeTarget = 50; } else { sizeTarget = 0; } current = p5.Vector.lerp(current, target, 0.1); sizeCurrent += (sizeTarget - sizeCurrent) * 0.2; p.ellipse(current.x, current.y, sizeCurrent); }; }; var myp5 = new p5(exampleOne, 'exampleOne');
